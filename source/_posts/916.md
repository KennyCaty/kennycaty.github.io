#include <stdio.h>
#include <stdlib.h>
#include <time.h>

//随机生成数字
int generateRandomNumber(int min, int max) {
    return rand() % (max - min + 1) + min;
}

//随机生成运算符
char getRandomop() {
    char ops[] = {'+', '-', '*', '/'};
    int index = generateRandomNumber(0, 3);
    return ops[index];
}

int main() {
    //设置随机数生成种子
    srand(time(NULL));

    int numQuestions;
    int numOps;

    printf("请输入题目数量：");
    fflush(stdout);
    scanf("%d", &numQuestions);

    printf("请输入每道题目的运算符数量：");
    fflush(stdout);
    scanf("%d", &numOps);

    //循环生成numQuestions道题目
    for (int i = 0; i < numQuestions; i++) {
        double operand1 = (double )generateRandomNumber(-100,100);
        printf("%lf", operand1);

        for (int j = 0; j < numOps; j++) {
            char op = getRandomop();
            double operand2 = (double )generateRandomNumber(-100,100);
            //op2必须是非零
            while (!operand2){
                operand2 = (double )generateRandomNumber(-100,100);
            }
            printf(" %c %lf", op, operand2);

            if (op == '+') {
                operand1 += operand2;
            } else if (op == '-') {
                operand1 -= operand2;
            } else if (op == '*') {
                operand1 *= operand2;
            } else if (op == '/') {
                operand1 /= operand2;
            }
        }
        printf(" = %lf\n", operand1);
    }

    return 0;
}
